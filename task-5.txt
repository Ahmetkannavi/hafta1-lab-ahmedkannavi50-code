    Yaz("Kahve türünü seçiniz (Filtre, Espresso, Turk, vb.): ")

        KahveMiktari ← 15 gram

    EĞER KahveTuru = "Espresso" İSE

    EĞER KahveTuru = "Turk" İSE

    Yaz("Kahve hazırlanıyor...")

    // Su sıcaklığını kontrol et
    EĞER SuSicakligi < 85 İSE

    DenemeSayisi ← 1
    TatSkoru ← 0

    // Kahve demleme döngüsü (tat tat ayarlama)
    WHILE TatSkoru < 8 VE DenemeSayisi ≤ 3 YAP
        Yaz("Demleme denemesi #" + DenemeSayisi)

        Yaz("Kahve öğütülüyor ve hazırlanıyor...")
        Yaz("Su " + SuSicakligi + "°C sıcaklığa ulaştı.")
        Yaz("Demleme süresi: " + DemlemeSuresi + " dakika")

        Yaz("Demleme işlemi devam ediyor...")
        Bekle(DemlemeSuresi dakika)

        Yaz("Kahve hazır! Tadım yapılıyor...")

        TatSkoru ← TatDegerlendir()  // Kullanıcı girdisi (örneğin 1–10 arası)
        EĞER TatSkoru < 8 İSE
            Yaz("Lezzet tam değil, ayarlama yapılıyor...")
            KahveMiktari ← KahveMiktari + 1
            SuSicakligi ← SuSicakligi + 1
        DEĞİLSE
            Yaz("İdeal kahve bulundu! ☕ Skor: " + TatSkoru)
            ÇIKIŞ
        SON

        DenemeSayisi ← DenemeSayisi + 1
    SON

    EĞER TatSkoru < 8 İSE
        Yaz("3 deneme sonunda ideal lezzet bulunamadı. Farklı kahve türü önerilir.")
